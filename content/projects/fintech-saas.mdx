async processPayment(data: TransactionPayload) {
  const isCompliant = await this.complianceEngine.check(data);
  if (!isCompliant) {
    throw new ComplianceError("Transaction flagged for review");
  }
  return await this.db.transaction(async (trx) => {
    const record = await trx.insert(data).into("ledgers");
    await this.cache.invalidate(`user:${data.userId}:balance`);
    return record;
  });
}
